278 HAL_StatusTypeDef HAL_InitTick (uint32_t TickPriority)
279 {
280  RCC_ClkInitTypeDef clkconfig;
281  uint32_t uwPrescalerValue = 0;
282  uint32_t pFLatency;
283 
284  /*Configure the TIMx IRQ priority */
285  HAL_NVIC_SetPriority(TIMx_IRQn, TickPriority);
286 
287  /* Enable the TIMx global Interrupt */
288  HAL_NVIC_EnableIRQ(TIMx_IRQn);
289 
290  /* Enable TIMx clock */
291  EnableClock_TIMx();
292 
293  /* Get clock configuration */
294  HAL_RCC_GetClockConfig(&clkconfig, &pFLatency);
295 
296  /* Compute the prescaler value to have TIM1 counter clock equal to 1MHz */
297  uwPrescalerValue = (uint32_t) ((HAL_TIM_GetPeriphClock(TIMx) / 1000000) - 1);
298 
299  /* Initialize TIMx peripheral as follow:
300  + Prescaler = (HAL_TIM_GetPeriphClock(TIMx)/1000000 - 1) to have a 1MHz TIM1 counter clock.
301  + Period = (TIM1 counter clock / TIM1 output clock) - 1. to have a (1/1000) s time base.
302  + ClockDivision = 0
303  + Counter direction = Up
304  */
305  htimx.Instance = TIMx;
306  htimx.Init.Period = (1000000 / 1000) - 1;
307  htimx.Init.Prescaler = uwPrescalerValue;
308  htimx.Init.ClockDivision = TIM_CLOCKDIVISION_DIV1;
309  htimx.Init.CounterMode = TIM_COUNTERMODE_UP;
310  if(HAL_TIM_Base_Init(&htimx) != HAL_OK)
311  {
312  /* Initialization Error */
313  Error_Handler();
314  }
315 
316  /* Start the TIM time Base generation in interrupt mode */
317  if(HAL_TIM_Base_Start_IT(&htimx) != HAL_OK)
318  {
319  /* Starting Error */
320  Error_Handler();
321  }
322 
323  /* Return function status */
324  return HAL_OK;
325 }
