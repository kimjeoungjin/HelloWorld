249 int main(void)
250 {
251  /* System initialization function */
252  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
253  {
254  /* Error during system clock configuration take appropriate action */
255  while(1);
256  }
257 
258  /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
259  HAL_Init();
260 
261  /* Configure leds */
262  BSP_LED_Init(BSP_LED1);
263  BSP_LED_Init(BSP_LED2);
264  BSP_LED_Init(BSP_LED3);
265 
266  /* Initialize all configured peripherals */
267  MX_GPIO_Init();
268  MX_USART1_UART_Init();
269 
270  /*-2- Configure UART peripheral for reception process (using LL) */
271  /* Any data received will be stored "aRxBuffer" buffer : the number max of
272  data received is RXBUFFERSIZE */
273  /* Enable RXNE and Error interrupts */
274  LL_USART_EnableIT_RXNE(USART1);
275  LL_USART_EnableIT_ERROR(USART1);
276 
277  /*-3- Start the transmission process (using LL) **/
278  /* While the UART in reception process, user can transmit data from
279  "aTxStartMessage" buffer */
280  /* Start USART transmission : Will initiate TXE interrupt after TDR register is empty */
281  LL_USART_TransmitData8(USART1, aTxStartMessage[uwTxIndex++]);
282 
283  BSP_LED_On(BSP_LED1);
284 
285  /* Enable TXE interrupt */
286  LL_USART_EnableIT_TXE(USART1);
287 
288  /*-4- Wait for the end of the transfer */
289  /* USART IRQ handler is not anymore routed to HAL_UART_IRQHandler() function
290  and is now based on LL API functions use.
291  Therefore, use of HAL IT based services is no more possible. */
292  /* Once TX transfer is completed, LED1 will turn On.
293  Then, when RX transfer is completed, LED2 will turn On. */
294  while (ubTxComplete == 0)
295  {
296  }
297 
298  while (ubRxComplete == 0)
299  {
300  }
301 
302  BSP_LED_Off(BSP_LED1);
303  BSP_LED_On(BSP_LED2);
304 
305 
306  /*-5- Send the received Buffer */
307  /* Even if use of HAL IT based services is no more possible, use of HAL Polling based services
308  (as Transmit in polling mode) is still possible. */
309  if(HAL_UART_Transmit(&huart1, (uint8_t*)aRxBuffer, RXBUFFERSIZE, 1000)!= HAL_OK)
310  {
311  /* Transfer error in transmission process */
312  Error_Handler();
313  }
314 
315  /*-6- Send the End Message */
316  if(HAL_UART_Transmit(&huart1, (uint8_t*)aTxEndMessage, TXENDMESSAGESIZE, 1000)!= HAL_OK)
317  {
318  /* Transfer error in transmission process */
319  Error_Handler();
320  }
321 
322  /* Infinite loop */
323  while (1)
324  {
325  }
326 }
