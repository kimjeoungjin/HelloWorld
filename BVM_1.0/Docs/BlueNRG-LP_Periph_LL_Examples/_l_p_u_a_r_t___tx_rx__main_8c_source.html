241 int main(void)
242 {
243  /* System initialization function */
244  if (SystemInit(SYSCLK_32M, RADIO_SYSCLK_NONE) != SUCCESS)
245  {
246  /* Error during system clock configuration take appropriate action */
247  while(1);
248  }
249 
250  /* Set systick to 1ms using system clock frequency */
251  LL_Init1msTick(SystemCoreClock);
252 
253  /* Initialize LED2 */
254  LED_Init();
255 
256  /* LED off */
257  LED_Off();
258 
259  /* Configure LPUART1 (LPUART IP configuration and related GPIO initialization) */
260  Configure_LPUART1();
261 
262  /* Start main program loop :
263  - Wait for any character received on LPUART RX line
264  */
265 
266  /* Send Text Information on LPUART TX to PC Com port */
267  PrintInfo();
268 
269  /* Infinite loop */
270  while (1)
271  {
272  if(newCharReceived == 1)
273  {
274  /* Check if received value is corresponding to specific one : S or s */
275  if((ubReceivedChar == 'S') || (ubReceivedChar == 's'))
276  {
277  /* Turn LED2 on : Expected character has been received */
278  LED_On();
279  }
280  else
281  {
282  /* Turn LED2 off */
283  LED_Off();
284  }
285 
286  /* Echo received character on TX */
287  LL_LPUART_TransmitData8(LPUART1, ubReceivedChar);
288 
289  /* Wait for TC flag to be raised for last char */
290  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
291  {
292  }
293 
294  newCharReceived = 0;
295  }
296  }
297 }

370 void PrintInfo(void)
371 {
372  uint32_t index = 0;
373 
374  /* Send characters one per one, until last char to be sent */
375  for (index = 0; index < sizeof(aTextInfo); index++)
376  {
377  /* Wait for TXE flag to be raised */
378  while (!LL_LPUART_IsActiveFlag_TXE(LPUART1))
379  {
380  }
381  /* Write character in Transmit Data register.
382  TXE flag is cleared by writing data in TDR register */
383  LL_LPUART_TransmitData8(LPUART1, aTextInfo[index]);
384  }
385 
386  /* Wait for TC flag to be raised for last char */
387  while (!LL_LPUART_IsActiveFlag_TC(LPUART1))
388  {
389  }
390 }
