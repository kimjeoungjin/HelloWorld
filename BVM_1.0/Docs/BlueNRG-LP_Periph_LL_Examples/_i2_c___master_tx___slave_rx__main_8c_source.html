516 void Handle_I2C_Slave(void)
517 {
518  /* (1) Wait ADDR flag and check address match code and direction ************/
519  while(!LL_I2C_IsActiveFlag_ADDR(I2Cx))
520  {
521  }
522 
523  /* Verify the Address Match with the OWN Slave address */
524  if(LL_I2C_GetAddressMatchCode(I2Cx) == SLAVE_OWN_ADDRESS)
525  {
526  /* Verify the transfer direction, a write direction, Slave enters receiver mode */
527  if(LL_I2C_GetTransferDirection(I2Cx) == LL_I2C_DIRECTION_WRITE)
528  {
529  /* Clear ADDR flag value in ISR register */
530  LL_I2C_ClearFlag_ADDR(I2Cx);
531  }
532  else
533  {
534  /* Clear ADDR flag value in ISR register */
535  LL_I2C_ClearFlag_ADDR(I2Cx);
536 
537  /* Call Error function */
538  Error_Callback();
539  }
540  }
541  else
542  {
543  /* Clear ADDR flag value in ISR register */
544  LL_I2C_ClearFlag_ADDR(I2Cx);
545 
546  /* Call Error function */
547  Error_Callback();
548  }
549 
550  /* (2) Loop until end of transfer received (STOP flag raised) ***************/
551 
552 #if (USE_TIMEOUT == 1)
553  Timeout = I2C_SEND_TIMEOUT_RXNE_MS;
554 #endif /* USE_TIMEOUT */
555 
556  /* Loop until STOP flag is raised */
557  while(!LL_I2C_IsActiveFlag_STOP(I2Cx))
558  {
559  /* (2.1) Receive data (RXNE flag raised) **********************************/
560 
561  /* Check RXNE flag value in ISR register */
562  if(LL_I2C_IsActiveFlag_RXNE(I2Cx))
563  {
564  /* Read character in Receive Data register.
565  RXNE flag is cleared by reading data in RXDR register */
566  aReceiveBuffer[ubReceiveIndex++] = LL_I2C_ReceiveData8(I2Cx);
567 
568 #if (USE_TIMEOUT == 1)
569  Timeout = I2C_SEND_TIMEOUT_RXNE_MS;
570 #endif /* USE_TIMEOUT */
571  }
572 
573 #if (USE_TIMEOUT == 1)
574  /* Check Systick counter flag to decrement the time-out value */
575  if (LL_SYSTICK_IsActiveCounterFlag())
576  {
577  if(Timeout-- == 0)
578  {
579  /* Time-out occurred. Set LED2 to blinking mode */
580  LED_Blinking(LED_BLINK_SLOW);
581  }
582  }
583 #endif /* USE_TIMEOUT */
584  }
585 
586  /* (3) Clear pending flags, Check Data consistency **************************/
587 
588  /* End of I2C_SlaveReceiver_MasterTransmitter_DMA Process */
589  LL_I2C_ClearFlag_STOP(I2Cx);
590 
591  /* Check if datas request to turn on the LED2 */
592  if(Buffercmp8((uint8_t*)aReceiveBuffer, (uint8_t*)aLedOn, (ubReceiveIndex-1)) == 0)
593  {
594  /* Turn LED2 On:
595  * - Expected bytes have been received
596  * - Slave Rx sequence completed successfully
597  */
598  LED_On();
599  printf("Slave Rx sequence completed successfully\n\r");
600  }
601  else
602  {
603  /* Call Error function */
604  Error_Callback();
605  }
606 }

663 void Handle_I2C_Master(void)
664 {
665  /* (1) Initiate a Start condition to the Slave device ***********************/
666 
667  /* Master Generate Start condition for a write request : */
668  /* - to the Slave with a 7-Bit SLAVE_OWN_ADDRESS */
669  /* - with a auto stop condition generation when transmit all bytes */
670  LL_I2C_HandleTransfer(I2Cx, SLAVE_OWN_ADDRESS, LL_I2C_ADDRSLAVE_7BIT, ubNbDataToTransmit, LL_I2C_MODE_AUTOEND, LL_I2C_GENERATE_START_WRITE);
671 
672  /* (2) Loop until end of transfer received (STOP flag raised) ***************/
673 
674 #if (USE_TIMEOUT == 1)
675  Timeout = I2C_SEND_TIMEOUT_TXIS_MS;
676 #endif /* USE_TIMEOUT */
677 
678 
679  /* Loop until STOP flag is raised */
680  while(!LL_I2C_IsActiveFlag_STOP(I2Cx))
681  {
682  /* (2.1) Transmit data (TXIS flag raised) *********************************/
683 
684  /* Check TXIS flag value in ISR register */
685  if(LL_I2C_IsActiveFlag_TXIS(I2Cx))
686  {
687  /* Write data in Transmit Data register.
688  TXIS flag is cleared by writing data in TXDR register */
689  LL_I2C_TransmitData8(I2Cx, (*pTransmitBuffer++));
690 
691 #if (USE_TIMEOUT == 1)
692  Timeout = I2C_SEND_TIMEOUT_TXIS_MS;
693 #endif /* USE_TIMEOUT */
694  }
695 
696 #if (USE_TIMEOUT == 1)
697  /* Check Systick counter flag to decrement the time-out value */
698  if (LL_SYSTICK_IsActiveCounterFlag())
699  {
700  if(Timeout-- == 0)
701  {
702  /* Time-out occurred. Set LED2 to blinking mode */
703  LED_Blinking(LED_BLINK_SLOW);
704  }
705  }
706 #endif /* USE_TIMEOUT */
707  }
708 
709  /* (3) Clear pending flags, Data consistency are checking into Slave process */
710 
711  /* End of I2C_SlaveReceiver_MasterTransmitter Process */
712  LL_I2C_ClearFlag_STOP(I2Cx);
713 
714  /* Turn LED2 On:
715  * - Expected bytes have been sent
716  * - Master Tx sequence completed successfully
717  */
718  LED_On();
719  printf("Master Tx sequence completed successfully\n\r");
720 
721  if(LL_I2C_IsActiveFlag_NACK(I2Cx))
722  {
723  printf("NACK is received after a byte transmission.\n\r");
724  }
725 }
