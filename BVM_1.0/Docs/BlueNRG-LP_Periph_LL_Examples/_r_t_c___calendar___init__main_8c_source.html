306 static void MX_RTC_Init(void)
307 {
308  LL_RTC_InitTypeDef RTC_InitStruct = {0};
309  LL_RTC_TimeTypeDef RTC_TimeStruct = {0};
310  LL_RTC_DateTypeDef RTC_DateStruct = {0};
311 
312  /* Check if Data stored in BackUp register1: No Need to reconfigure RTC */
313  /* Read the Back Up Register 1 Data */
314  if (LL_RTC_BAK_GetRegister(RTC, LL_RTC_BKP_DR1) != RTC_BKP_DATE_TIME_UPDTATED)
315  {
316  printf("Check Data stored in BackUp register1: configuration RTC started\n\r");
317  /* Force RTC peripheral reset */
318  /* This action delete the calendar configuration in case of hardware reset */
319 
320  /* Enable Peripheral Clock */
321  LL_APB0_EnableClock(LL_APB0_PERIPH_RTC);
322 
323  /* RTC peripheral reset */
324  LL_RCC_ClearFlag_RTCRSTREL();
325  LL_APB0_ForceReset(LL_APB0_PERIPH_RTC);
326  LL_APB0_ReleaseReset(LL_APB0_PERIPH_RTC);
327  while(LL_RCC_IsActiveFlag_RTCRSTREL() == 0);
328  LL_RCC_ClearFlag_RTCRSTREL();
329 
330  /* Disable the write protection for RTC registers */
331  LL_RTC_DisableWriteProtection(RTC);
332 
333  /* Init mode setup */
334  LL_RTC_EnableInitMode(RTC);
335 
336  /* Wait till the Init mode is active */
337  while(LL_RTC_IsActiveFlag_INIT(RTC) == RESET);
338 
339  /* Enter in initialization mode */
340  if (Enter_RTC_InitMode() != RTC_ERROR_NONE)
341  {
342  /* Initialization Error */
343  LED_Blinking(LED_BLINK_ERROR);
344  }
345 
346  /* Configure RTC */
347  RTC_InitStruct.HourFormat = LL_RTC_HOURFORMAT_24HOUR;
348  RTC_InitStruct.AsynchPrescaler = RTC_ASYNCH_PREDIV;
349  RTC_InitStruct.SynchPrescaler = RTC_SYNCH_PREDIV;
350  LL_RTC_Init(RTC, &RTC_InitStruct);
351 
352  /* Configure the Time */
353  RTC_TimeStruct.TimeFormat = LL_RTC_TIME_FORMAT_AM_OR_24;
354  RTC_TimeStruct.Hours = 0x23;
355  RTC_TimeStruct.Minutes = 0x59;
356  RTC_TimeStruct.Seconds = 0x50;
357  LL_RTC_TIME_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_TimeStruct);
358  /* Configure the Date */
359  RTC_DateStruct.WeekDay = LL_RTC_WEEKDAY_FRIDAY;
360  RTC_DateStruct.Month = LL_RTC_MONTH_DECEMBER;
361  RTC_DateStruct.Day = 0x31;
362  RTC_DateStruct.Year = 0x19;
363  LL_RTC_DATE_Init(RTC, LL_RTC_FORMAT_BCD, &RTC_DateStruct);
364 
365  /* Exit of initialization mode */
366  if (Exit_RTC_InitMode() != RTC_ERROR_NONE)
367  {
368  /* Initialization Error */
369  LED_Blinking(LED_BLINK_ERROR);
370  }
371 
372  /* Enable RTC registers write protection */
373  LL_RTC_EnableWriteProtection(RTC);
374 
375  /* Writes a data in a RTC Backup data Register1 */
376  LL_RTC_BAK_SetRegister(RTC, LL_RTC_BKP_DR1, RTC_BKP_DATE_TIME_UPDTATED);
377  }
378  else
379  {
380  printf("Check Data stored in BackUp register1: No Need to reconfigure RTC\n\r");
381  }
382 }
